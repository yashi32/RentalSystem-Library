<h1>Library Rental System</h1>
<h2>Objective of assignment</h2>
<ul>
    <li>Create a Book</li>
        <li>Create a User</li>
            <li>Rent a Book</li>
                <li>List all Rented Books</li>



</ul>

<h2> Backend</h2>
<ul>
        <li>Node.js</li>
        <li>Express</li>
        <li>MongoDB</li>
</ul>

<h2> Dependicies</h2>
<ul>
        <li>npm install express</li>
        <li>npm install mongoose </li>
        <li>npm install</li>
</ul>

<h2> Start the app</h2>
<ul>
        <li>node index.js</li>
        
</ul>

<h2> API Endpoints</h2>
<ul>
        <li>POST 'library/create-book'</li>
        <li>POST 'library/create-user' </li>
        <li>POST 'library/rent-book'</li>
          <li>GET 'library/rented-books-lists'</li>
</ul>
